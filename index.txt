function calcType(answers){
  // Q1-3: 추리 스타일 (A=0논리, B=1균형, C=2직감)
  // Q4-6: 게임 태도 (A=0주도, B=1협력, C=2잠행)
  const styleMap={A:0,B:1,C:2};
  const h=[0,0,0],v=[0,0,0];
  [0,1,2].forEach(i=>{ if(answers[i]) h[styleMap[answers[i]]]++; });
  [3,4,5].forEach(i=>{ if(answers[i]) v[styleMap[answers[i]]]++; });
  const hIdx=h.indexOf(Math.max(...h));
  const vIdx=v.indexOf(Math.max(...v));
  return Object.entries(TYPES).find(([,t])=>t.axis[0]===hIdx&&t.axis[1]===vIdx)?.[0]||"균형의 조율가";
}

// ── API 함수 ──────────────────────────────────────
async function gs(action,data=null,id=null,prompt=null){
  const url=new URL(API_URL);
  url.searchParams.set("action",action);
  if(data) url.searchParams.set("data",JSON.stringify(data));
  if(id!==null) url.searchParams.set("id",String(id));
  if(prompt) url.searchParams.set("prompt",prompt);
  const r=await fetch(url.toString());
  const j=await r.json();
  if(!j.ok) throw new Error(j.error);
  return j.data;
}
async function gemini(prompt){
  return await gs("callGemini",null,null,prompt);
}

// ── 앱 루트 ───────────────────────────────────────
function App(){
  const [me,setMe]=useState(()=>{
    try{const s=localStorage.getItem("mm_me");return s?JSON.parse(s):null;}catch{return null;}
  });
  const [page,setPage]=useState(()=>{
    try{return localStorage.getItem("mm_page")||"dash";}catch{return "dash";}
  });
  const [auth,setAuth]=useState("login");
  const [mbs,setMbs]=useState([]);
  const [themes,setThemes]=useState([]);
  const [records,setRecords]=useState([]);
  const [schedules,setSchedules]=useState([]);
  const [modal,setModal]=useState(null);

  const goPage=p=>{setPage(p);try{localStorage.setItem("mm_page",p);}catch{}};
  const close=()=>setModal(null);
  const pending=themes.filter(t=>!t.ok&&t.status==="pending");

  useEffect(()=>{
    gs("getMembers").then(d=>{if(d?.length)setMbs(d);}).catch(()=>{});
  },[]);

  useEffect(()=>{
    if(!me) return;
    gs("getThemes").then(d=>{if(d) setThemes(d.map(safeTheme));}).catch(()=>{});
    gs("getRecords").then(d=>{if(d) setRecords(d);}).catch(()=>{});
    gs("getSchedules").then(d=>{if(d) setSchedules(d);}).catch(()=>{});
  },[me]);

  function safeTheme(t){
    return{...t,
      ok:t.ok===true||t.ok==="TRUE"||t.ok==="true",
      characters:parseArr(t.characters),
      tags:parseArr(t.tags,true),
      diff:Number(t.diff)||3,
      time:Number(t.time)||120,
      rating:t.rating?Number(t.rating):null,
    };
  }
  function parseArr(v,csv=false){
    if(Array.isArray(v)) return v;
    if(typeof v==="string"&&v.trim()){
      if(csv) return v.split(",").map(s=>s.trim()).filter(Boolean);
      try{return JSON.parse(v);}catch{}
    }
    return[];
  }

  const login=m=>{
    setMe(m);
    try{localStorage.setItem("mm_me",JSON.stringify(m));}catch{}
    goPage("dash");
  };
  const logout=()=>{
    setMe(null);
    try{localStorage.removeItem("mm_me");localStorage.removeItem("mm_page");}catch{}
  };

  if(!me){
    if(auth==="signup") return(
      <Signup
        onDone={m=>{setMbs(p=>[...p,m]);login(m);setAuth("login");}}
        onBack={()=>setAuth("login")}
        existingIds={mbs.map(m=>m.id).filter(Boolean).map(Number)}
      />
    );
    return(
      <Login mbs={mbs} onLogin={login} onSignup={()=>setAuth("signup")}
        onRefresh={setMbs}/>
    );
  }

  const props={me,mbs,setMbs,themes,setThemes,records,setRecords,
    schedules,setSchedules,goPage,modal,setModal,close};

  return(
    <div style={{display:"flex",minHeight:"100vh",background:C.bg}}>
      <Sidebar page={page} setPage={goPage} me={me} badge={me?.role==="admin"?pending.length:0} onLogout={logout}/>
      <main style={{flex:1,overflowY:"auto",padding:"32px 36px",minWidth:0}}>
        {page==="dash"  &&<Dash {...props}/>}
        {page==="my"    &&<MyPage {...props}/>}
        {page==="fri"   &&<Friends {...props}/>}
        {page==="crew"  &&<Crew {...props}/>}
        {page==="theme" &&<ThemeDB {...props}/>}
        {page==="sch"   &&<Schedule {...props}/>}
        {page==="rec"   &&<AIRec {...props}/>}
        {page==="hof"   &&<HoF {...props}/>}
      </main>
      {modal&&<ModalRouter {...props}/>}
    </div>
  );
}

function ModalRouter(props){
  const{modal,close}=props;
  if(modal?.t==="addRecord")  return <Modal title="기록 추가" onClose={close}><RecordForm {...props}/></Modal>;
  if(modal?.t==="editRecord") return <Modal title="기록 수정" onClose={close}><RecordForm init={modal.d} {...props}/></Modal>;
  if(modal?.t==="addTheme")   return <Modal title="테마 제보" onClose={close} wide><ThemeForm {...props}/></Modal>;
  if(modal?.t==="addSch")     return <Modal title="일정 등록" onClose={close}><SchForm {...props}/></Modal>;
  if(modal?.t==="addCrew")    return <Modal title="모임 만들기" onClose={close}><CrewForm {...props}/></Modal>;
  if(modal?.t==="themeDetail")return <Modal title={modal.d?.title} onClose={close} wide><ThemeDetail d={modal.d} {...props}/></Modal>;
  return null;
}

function Friends({me,mbs,setMbs,records}){

  const [searchResult,setSearchResult]=useState(null);
  const [sel,setSel]=useState(null);

    let fl=me.friends;
    if(typeof fl==="string"){try{fl=JSON.parse(fl);}catch{fl=[];}}
    if(!Array.isArray(fl)) fl=[];
    return mbs.filter(m=>fl.map(String).includes(String(m.id)));
  })();

  const addFri=async(fid)=>{
    const fl=myFriends.map(m=>m.id);
    if(fl.includes(fid)) return;
    const newFl=[...fl,fid];
    const updated={...me,friends:JSON.stringify(newFl)};
    try{await gs("updateMember",updated);}catch{}
    setMbs(p=>p.map(m=>String(m.id)===String(me.id)?{...m,friends:JSON.stringify(newFl)}:m));
  };

    if(!search.trim()){setSearchResult("empty");return;}

    setSearchResult(found||"notfound");

  const compat=sf?getCompat(me.type,sf.type):null;

    if(!a||!b) return null;
    const ta=TYPES[a],tb=TYPES[b];
    if(!ta||!tb) return null;
    const dx=Math.abs(ta.axis[0]-tb.axis[0]);
    const dy=Math.abs(ta.axis[1]-tb.axis[1]);
    const dist=dx+dy;
    const score=Math.max(40,100-dist*15);

    const color=score>=85?"#059669":score>=70?"#2563EB":score>=55?"#D97706":"#DC2626";
    return{score,label,color};
  }

  const shared=sf?records.filter(r=>{
    const mbs2=Array.isArray(r.mbs)?r.mbs.map(String):[];
    return mbs2.includes(String(me.id))&&mbs2.includes(String(sf.id));

{"score":85,"chemistry":"...","recommend":"..."}`;

// ── AI 추천 ───────────────────────────────────────
function AIRec({me,themes,records}){
  const [rec,setRec]=useState(null);
  const [loading,setLoading]=useState(false);
  const today=new Date().toISOString().slice(0,10);
  const cacheKey=`mm_airec_${me.id}_${today}`;

  useEffect(()=>{
    try{
      const cached=localStorage.getItem(cacheKey);
      if(cached){setRec(JSON.parse(cached));return;}
    }catch{}
    getAIRec();
  },[]);

  const getAIRec=async()=>{
    setLoading(true);
    const played=records.filter(r=>{
      const mbs=Array.isArray(r.mbs)?r.mbs.map(String):[];
      return mbs.includes(String(me.id));
    }).map(r=>r.themeTitle).filter(Boolean);
    const unplayed=themes.filter(t=>t.ok&&!played.includes(t.title));
    if(unplayed.length===0){setRec({text:"모든 테마를 플레이하셨네요! 새 테마를 제보해주세요."});setLoading(false);return;}
    const prompt=`머더미스터리 탐정 ${me.name}님 (${me.type||"유형 미설정"}, 강점: ${me.strength||"미설정"})에게 오늘의 추천 테마.

플레이 가능한 테마: ${unplayed.slice(0,5).map(t=>t.title).join(", ")||"없음"}

한 가지를 추천하고 그 이유를 2-3문장으로 설명해줘. 친근한 말투로.`;
    try{
      const txt=await gemini(prompt);
      const result={text:txt,date:today};
      setRec(result);
      try{localStorage.setItem(cacheKey,JSON.stringify(result));}catch{}
    }catch{setRec({text:"AI 추천을 불러오지 못했어요. 잠시 후 다시 시도해주세요."});}
    setLoading(false);
  };

  const myR=records.filter(r=>{
    const mbs=Array.isArray(r.mbs)?r.mbs.map(String):[];
    return mbs.includes(String(me.id));
  });
  const unplayedThemes=themes.filter(t=>t.ok&&!myR.map(r=>r.themeTitle).includes(t.title));

  return(
    <div className="fade">
      <h1 style={{fontSize:24,fontWeight:900,marginBottom:20}}>AI 추천</h1>
      <Card style={{marginBottom:20}}>
        <div style={{fontSize:13,fontWeight:700,color:C.gold,marginBottom:12}}>🤖 오늘의 맞춤 추천</div>
        {loading&&<div style={{color:C.muted,fontSize:13}}>AI가 분석 중...</div>}
        {rec&&<div style={{fontSize:13,color:C.text,lineHeight:1.8}}>{rec.text}</div>}
        <div style={{marginTop:12}}>
          <Btn sm outline onClick={getAIRec} disabled={loading}>다시 추천받기</Btn>
        </div>
      </Card>
      <div style={{fontSize:14,fontWeight:700,marginBottom:12}}>미플레이 테마 ({unplayedThemes.length})</div>
      <div style={{display:"flex",flexDirection:"column",gap:9}}>
        {unplayedThemes.slice(0,5).map(t=>(
          <Card key={t.id} style={{display:"flex",justifyContent:"space-between",alignItems:"center"}}>
            <div>
              <div style={{fontSize:14,fontWeight:600}}>{t.title}</div>
              <div style={{fontSize:11,color:C.muted,marginTop:2}}>{t.theme} · {t.time}분</div>
            </div>
            <span style={{fontSize:11,color:C.gold,fontWeight:600}}>추천 ↑</span>
          </Card>
        ))}
      </div>
    </div>
  );
}

// ── 명예의 전당 ───────────────────────────────────
function HoF({mbs,records}){
  const stats=mbs.map(m=>{
    const myR=records.filter(r=>{
      const mb=Array.isArray(r.mbs)?r.mbs.map(String):[];
      return mb.includes(String(m.id));
    });

    const wins=myR.filter(r=>r.res==="성공").length;
    const rate=myR.length?Math.round(wins/myR.length*100):0;
    return{...m,games:myR.length,mvps,wins,rate};
